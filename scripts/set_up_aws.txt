# Connect to aws instance
chmod 400 ~/tutorial.pem
ssh -i "~/tutorial.pem" ec2-user@ec2-3-9-132-171.eu-west-2.compute.amazonaws.com

# Update packages
sudo yum update # sudo apt-get update for ubuntu instance
sudo yum install python3
sudo yum install git

# Clone CIFO_project repository
git clone https://github.com/DavidSilva98/CIFO_project.git
cd CIFO_project

# Create and set-up environment
python3 -m venv .env
source .env/bin/activate
pip install -r requirements.txt

# Run & Create logs
python ga_script_TSP_data.py

# Export data to local machine
scp -i "~/tutorial.pem" ec2-user@ec2-3-9-132-171.eu-west-2.compute.amazonaws.com:~/CIFO_project/log_all.zip D:/Downloads
